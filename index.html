if(PlayerHeroASComponent) //The ability system component
{
	
	//Check if ability is ultimate and if that ability level is 3
	if(GetAbilityTreeType(InInputSlot).MatchesTag(TAG_UI_Filter_UltimateAbility) &&
		GetAbilitySlotLevel(InInputSlot) == 3)
	{
		FString TagToAdd = "Blocked.";

		//Add our ability name dynamically
		TagToAdd.Append(GetAbilityClassName(InInputSlot));

		PlayerHeroASComponent->RemoveLooseGameplayTag(
			FGameplayTag::RequestGameplayTag(FName(TagToAdd))); 
	}

	const FGameplayAbilitySpec* AbilitySpec =
		PlayerHeroASComponent->FindAbilitySpecFromHandle(AbilitySlotSpecHandles[InInputSlot -1]);

	TArray GameplayAbilityInstances = AbilitySpec->GetAbilityInstances();

	//This will always be one, because we will never have multiple of this ability and we already know 
	//the ability exists
	if (GameplayAbilityInstances.Num() == 1)
	{
		//Looping through each value we want to retrieve to edit
		//Each path can have one or more values per a tree path, EX: Path 1 affects cooldown and damage
		// while path 2 only affects ability range
		for (int32 Index = 0; Index < GetPathArrayNum(InInputSlot, GetChosenPath(InInputSlot)); Index++)
		{
			UE_LOG(LogAven, Log, TEXT("Applied value %f to ability %s"),
				GetCurrentDesiredValueOfSlot(InInputSlot, Index), *GameplayAbilityInstances[0]->GetName());

			if(UHeroGameplayAbility* HeroGameplayAbility = Cast(
				GameplayAbilityInstances[0]))
			{
				HeroGameplayAbility->ApplyTreePathExecution(
					GetCurrentDesiredValueOfSlot(InInputSlot, Index), GetChosenPath(InInputSlot),
					InInputSlot, Index, this);
			}
		}
	}
}

In this example I'm modifying an instance of an ability every time I add points to that ability. From the top I'm dynamically removing a tag which was added dynamically when we initialized the ultimate ability. The ability settings are set to block that specific ability if the tag is active on the player's ability system component. Next I'm pulling from ability specs which I stored when the abilities were given. The if statement regarding the ability instance will never be more or less than 1, so that statement is for readability and clarity only. The for loop is there for editing multiple values within a tree path. The abilities in this game have two paths in which each path modifies the ability in some way. I designed the structure to be flexible and accommodate any number of changes.

void UAvenAssetManager::LoadCharacterData(const FString& InUniquePlayerID,const FCharacterData& InCharacterData, 
const FStreamableDelegate& LoadFinishedDelegateToCall, bool bLoadingFromPIE)
{
	TArray DataAssetsToLoad;

		//Retrieve all the assets we need to load for this character
		GetRelevantCharacterDataAssetIDs(InCharacterData, DataAssetsToLoad);
		
		if(bLoadingFromPIE)
		{
			const UAbilityDataLibrary* AbilityDL = AbilityDataLibrary;
			const UEquipmentTypeDataLibrary* EquipmentTypeDL = EquipmentTypeDataLibrary;
		
			bool bHaveAllAssets = EquipmentTypeDL && AbilityDL;
		
			if(bHaveAllAssets)
			{
				DataAssetsToLoad.Append(AbilityDL->Library);
				DataAssetsToLoad.Append(EquipmentTypeDL->GetAllTypesToLoad());
			}
	
			TArray Bundles;
			Bundles.Add(FName("Full"));
			Bundles.Add(FName("Info"));
		
			//Load character data assets and assign them a handle that can be used to access them
			//Allows us to cancel or check progress
			TSharedPtr LoadingHandle = LoadPrimaryAssets(DataAssetsToLoad, Bundles,
				LoadFinishedDelegateToCall, FStreamableManager::AsyncLoadHighPriority);
	
			//Keep track of how many assets of this type we have for each player and store handle
			PlayerLoadedAssetData.ReplaceOrAddPlayerLoadData(InUniquePlayerID, DataAssetsToLoad, LoadingHandle);
	
			return;
		}
	
		//Add assets we need for this character
		TArray BundlesToAdd;
		BundlesToAdd.Add(FName("Full"));
		
		TSharedPtr LoadingHandle = ChangeBundleStateForPrimaryAssets(DataAssetsToLoad, BundlesToAdd,
			TArray(), false, LoadFinishedDelegateToCall);
	
		//Keep track of how many assets of this type we have for each player and store handle
		PlayerLoadedAssetData.ReplaceOrAddPlayerLoadData(InUniquePlayerID, DataAssetsToLoad, LoadingHandle);
}

In the example above I've separated data from data assets into two categories: Full and Info. Info is information only, and Full is everything else. I created a struct that holds player data and counts the number of other plays who share the same asset. This is done to accurately remove assets when they are no longer being used.

void ACharLobbyController::SetCharMeshForPlayerState(AAvenLobbyPlayerState* InPS)
{
	if(InPS == nullptr || !IsLocalController())
	{
		return;
	}

	//Find and store pointers to the spawner for each team
	if (!TeamOnePawnSpawner || !TeamTwoPawnSpawner)
	{
		UWorld* World = GetWorld();

		for (TActorIterator It(World); It; ++It)
		{
			ALobbyPawnSpawner* CharLocations = *It;

			if (CharLocations)
			{
				if(CharLocations->GetTeamID() == 1)
				{
					TeamOnePawnSpawner = CharLocations;

					TeamOnePawnSpawner->SpawnMeshComponents();
				}
				else if (CharLocations->GetTeamID() == 2)
				{
					TeamTwoPawnSpawner = CharLocations;
					
					TeamTwoPawnSpawner->SpawnMeshComponents();
				}

				if (TeamOnePawnSpawner && TeamTwoPawnSpawner)
				{
					break;
				}
			}
		}
	}

	//Depending on whether we local or remote assign each of the character meshes to each player
	//Each mesh is handled locally and only information on characters are replicated
	//This information is used to set up the mesh, local owning player will always control center mesh
	if(InPS->GetLocalCharMesh() == nullptr)
	{
		if(InPS->GetGenericTeamId().GetId() == 1)
		{
			if(TeamOnePawnSpawner)
			{
				if(InPS->GetPlayerController() && InPS->GetPlayerController()->IsLocalController())
				{
					ACharLobbyMesh* LocalMesh = nullptr;
					
					if(TeamOnePawnSpawner->PlayerCharMeshByIndex(0))
					{
						LocalMesh = Cast(TeamOnePawnSpawner->PlayerCharMeshByIndex(0)->GetChildActor());
					}

					if(LocalMesh)
					{
						if(!LocalMesh->GetIsTaken())
						{
							InPS->SetLocalCharMesh(LocalMesh);

							LocalMesh->SetIsTaken(true);

							LocalMesh->SetOriginalStartingLocation(TeamOnePawnSpawner->GetCapsuleByIndex(0)->GetComponentLocation());
					
						}
					}
				}
				else
				{
					ACharLobbyMesh* Mesh = nullptr;
					
					if(TeamOnePawnSpawner->PlayerCharMeshByIndex(TeamOneIndexCount))
					{
						Mesh = Cast(TeamOnePawnSpawner->PlayerCharMeshByIndex(TeamOneIndexCount)->GetChildActor());
					}
					
					if(Mesh)
					{
						if(!Mesh->GetIsTaken())
						{
							InPS->SetLocalCharMesh(Mesh);

							Mesh->SetIsTaken(true);

							Mesh->SetOriginalStartingLocation(TeamOnePawnSpawner->GetCapsuleByIndex(TeamOneIndexCount)->GetComponentLocation());
			
							TeamOneIndexCount += 1;
						}
					}
				}
			}
		}
		else if (InPS->GetGenericTeamId().GetId() == 2)
		{
			if(TeamTwoPawnSpawner)
			{
				if(InPS->GetPlayerController() && InPS->GetPlayerController()->IsLocalController())
				{
					ACharLobbyMesh* LocalMesh = nullptr;
					
					if(TeamOnePawnSpawner->PlayerCharMeshByIndex(0))
					{
						LocalMesh = Cast(TeamTwoPawnSpawner->PlayerCharMeshByIndex(0)->GetChildActor());
					}
					
					if(LocalMesh)
					{
						if(!LocalMesh->GetIsTaken())
						{
							InPS->SetLocalCharMesh(LocalMesh);

							LocalMesh->SetIsTaken(true);

							LocalMesh->SetOriginalStartingLocation(TeamTwoPawnSpawner->GetCapsuleByIndex(0)->GetComponentLocation());
					
						}
					}
				}
				else
				{
					ACharLobbyMesh* Mesh = nullptr;
					
					if(TeamOnePawnSpawner->PlayerCharMeshByIndex(TeamTwoIndexCount))
					{
						Mesh = Cast(TeamTwoPawnSpawner->PlayerCharMeshByIndex(TeamTwoIndexCount)->GetChildActor());
					}
					
					if(Mesh)
					{
						if(!Mesh->GetIsTaken())
						{
							InPS->SetLocalCharMesh(Mesh);
	
							Mesh->SetIsTaken(true);

							Mesh->SetOriginalStartingLocation(TeamTwoPawnSpawner->GetCapsuleByIndex(TeamTwoIndexCount)->GetComponentLocation());
				
							TeamTwoIndexCount += 1;
						}
					}
				}
			}
		}
	}
}

This function is being fired on both server and clients. This game is currently using a listen server for testing multiplayer. Because a listen server also plays the game we are running this code on server as well. Two things to note with this code: It replicates as little as possible. The code itself is only being ran locally, but it is using replicated player character information to set up the mesh. Since the mesh exists on both server and client, the client can spawn and process the mesh locally using replicated player data. This allows us to be more flexible with the meshes, such as changing what player's mesh is spawned where. Because GetPlayerController does not exist on remote clients, and only exists on owning client and server, we know that the else will be triggered for everything remote players. IsLocalController ensures that regardless if player server or not, the controller we're accessing has to be local. This means that even though server has access to controller the remote host will trigger the else statement. Since we want owning clients to select and use the mesh at the center we default center mesh to index 0.

void AMinionAIControllerBase::SetMoveToLocation()
{
	const int32 SplineEnd = SplineActor->GetSpline()->GetNumberOfSplinePoints()  - 1;

	if (!bIsStartingAtEndOfSpline)
	{
		if (CurrentSplinePoint == SplineActor->GetSpline()->GetNumberOfSplinePoints() - 3)
		{	//If we're on the last overlap just make location end of spline
			MoveLocation = SplineActor->GetSpline()->GetLocationAtSplinePoint(SplineEnd, ESplineCoordinateSpace::World);
		}
		else
		{
			const float FirstPointDistance = SplineActor->GetSpline()->GetDistanceAlongSplineAtSplinePoint(NextSplinePoint);

			const float SecondPointDistance = SplineActor->GetSpline()->GetDistanceAlongSplineAtSplinePoint(NextSplinePoint + 1);

			const float MidPointOfSpline = (SecondPointDistance - FirstPointDistance) / 2;


			const FVector FirstPointLoc = SplineActor->GetSpline()->GetLocationAtSplinePoint(NextSplinePoint, ESplineCoordinateSpace::World);

			const FVector SecondPointLoc = SplineActor->GetSpline()->GetLocationAtSplinePoint(NextSplinePoint + 1, ESplineCoordinateSpace::World);

			const FVector Direction = (SecondPointLoc - FirstPointLoc).GetSafeNormal();


			//We're adding the midpoint location to location where the spline is calculating it
			//This gives us he mid point of the next overlap, taking it's size into consideration
			MoveLocation = FirstPointLoc + (Direction * MidPointOfSpline);
		}
	}
	else
	{
		if (CurrentSplinePoint == 1)
		{	//If we're on the last overlap just make location end of spline
			MoveLocation = SplineActor->GetSpline()->GetLocationAtSplinePoint(0, ESplineCoordinateSpace::World);
		}
		else
		{
			const float FirstPointDistance = SplineActor->GetSpline()->GetDistanceAlongSplineAtSplinePoint(NextSplinePoint);

			const float SecondPointDistance = SplineActor->GetSpline()->GetDistanceAlongSplineAtSplinePoint(NextSplinePoint - 1);

			const float MidPointOfSpline = (SecondPointDistance - FirstPointDistance) / 2;


			const FVector FirstPointLoc = SplineActor->GetSpline()->GetLocationAtSplinePoint(NextSplinePoint, ESplineCoordinateSpace::World);

			const FVector SecondPointLoc = SplineActor->GetSpline()->GetLocationAtSplinePoint(NextSplinePoint - 1, ESplineCoordinateSpace::World);

			const FVector Direction = (SecondPointLoc - FirstPointLoc).GetSafeNormal();


			//We're adding the midpoint location to location where the spline is calculating it
			//This gives us he mid point of the next overlap, taking it's size into consideration
			MoveLocation = FirstPointLoc + (Direction * MidPointOfSpline);
		}
	}
}

Using math and a custom spline tool I created that links overlap boxes with splines, this code uses an overlap to generate a direction which is then fed to minion AI to tell them where they need to go. Because each overlap holds information on each spline point the minions are aware of where they are on the spline at all times. NextSplinePoint is fed into the minions each time the minion overlaps with the spline. Though minions can switch between moving on spline, attacking player, other minions or buildings, the minion default behavior is to move on the spline and will return to spline when all other tasks are unavailable.

void APlayerHeroControllerBase::SetupInputComponent()
{
	Super::SetupInputComponent();
	
	if(UAvenEnhancedInputComponent* EnhancedInput = Cast(InputComponent))
	{
		EnhancedInput->BindNativeAction(InputConfig, TAG_MovementInput_Look_Mouse, ETriggerEvent::Triggered, this, &ThisClass::Input_LookMouse);
		EnhancedInput->BindNativeAction(InputConfig, TAG_MovementInput_Look_Stick, ETriggerEvent::Triggered, this, &ThisClass::Input_LookStick);
		EnhancedInput->BindNativeAction(InputConfig, TAG_UIInput_Game_Interact, ETriggerEvent::Triggered, this, &ThisClass::Input_Interact);
	}
}

void APlayerHeroControllerBase::Input_Interact(const FInputActionValue& InputActionValue)
{
	if(ATeleportStatue* Teleporter = Cast(InteractedActor))
	{
		if(UTeleporterWidget* TeleporterWidget = Teleporter->GetTeleportWidget())
		{
			Teleporter->DeactivateTeleporterWidget();
		}
		else
		{
			Teleporter->ActivateTeleporterWidget(this);
		}
	}
	else if(AShop* Shop = Cast(InteractedActor))
	{
		if(UItemShopWidget* ShopWidget = Shop->GetShopWidget())
		{
			Shop->DeactivateShopWidget();
		}
		else
		{
			Shop->ActivateShopWidget(this);
		}
	}
}

After binding new input system to C++ interact function, I'm both activating and deactivating the widget based on whether the widget exists on that actor. Each of these actors stores a pointer to the widget when it is pushed onto the stack, and that pointer is removed when it is removed from the stack. The pointer to that actor is being stored whenever the player overlaps with that actor, so we're only able to influence the UI while we're within range of the actor. Other code also removes UI when player leaves overlap. This turns a simple button press into something more like a toggle for the UI.
